# ==============================
# Core numerical / data stack
# ==============================

numpy~=2.0
# Numerical array foundation. Pinned to minor to avoid ABI breakage
# affecting scipy / pandas wheels.

pandas~=2.3
# Tabular data structures and DataFrame operations used throughout
# dataset metadata and aggregation layers.

scipy~=1.13
# Scientific routines (sparse matrices, stats) used by scanpy
# and downstream analytical helpers.


# ==============================
# Single-cell ecosystem
# ==============================

anndata~=0.10
# Core container for single-cell data (.X / .obs / .var / .obsm).
# Version-coupled with scanpy and numpy.

scanpy~=1.10
# High-level single-cell analysis toolkit.
# Used for filtering, QC, and expression summaries.


# ==============================
# Web UI (Dash)
# ==============================

dash~=3.3
# Dash web framework (Flask-based).
# Provides layout, callbacks, and app lifecycle.

dash-bootstrap-components~=2.0
# Bootstrap UI components and themes for Dash layouts.


# ==============================
# Plotting / export
# ==============================

plotly~=6.5
# Interactive plotting backend used by all views.

kaleido~=0.2
# Static image export engine for Plotly (PNG/SVG/PDF).
# Pinned to avoid runtime regressions in headless Docker environments.


# ==============================
# Server / logging
# ==============================

gunicorn~=22.0
# Production WSGI server used in Docker deployments.
# Avoids Dash's built-in dev server.

python-json-logger~=4.0
# Structured JSON logging for containerised environments and CI logs.
